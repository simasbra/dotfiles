let mapleader = " "
let maplocalleader = " "

set number
set relativenumber

" Enable mouse mode
set mouse=a

" Don't show the mode in the command line
set showmode=false

" Sync clipboard with the OS
set clipboard=unnamedplus

" Enable break indent
set breakindent

" Save undo history
set undofile

" Case-insensitive searching, unless \C or uppercase letters are used
set ignorecase
set smartcase

" Keep sign column on by default
set signcolumn=yes

" Decrease update time
set updatetime=250

" Decrease mapped sequence wait time
set timeoutlen=300

" Configure how new splits should be opened
set splitright
set splitbelow

" Show certain whitespace characters
set list
set listchars=tab:»\ ,trail:·,nbsp:␣

" Preview substitutions live
set inccommand=split

" Highlight the line with the cursor
set cursorline

" Minimal number of screen lines to keep above and below the cursor
set scrolloff=10

set tabstop=8
set shiftwidth=8
set softtabstop=8
set noexpandtab

nnoremap <C-d> <C-d>zz " Move down a page
nnoremap <C-u> <C-u>zz " Move up a page

" Window remappings
nnoremap <C-w>h <C-w>h " Go to the left window
nnoremap <C-w>n <C-w>j " Go to the down window
nnoremap <C-w>l <C-w>k " Go to the up window
nnoremap <C-w>e <C-w>l " Go to the right window
nnoremap <C-w><C-h> <C-w>h " Go to the left window
nnoremap <C-w><C-n> <C-w>j " Go to the down window
nnoremap <C-w><C-l> <C-w>k " Go to the up window
nnoremap <C-w><C-e> <C-w>l " Go to the right window

" Control remappings
nnoremap J E " Jump end WORD (swap E)
nnoremap j e " Jump end word (swap e)
nnoremap N K " Help (swap K)
nnoremap n j " Down (swap j)
nnoremap E J " Join lines (swap J)
nnoremap e l " Right (swap l)
nnoremap K Nzz " Previous find (swap N)
nnoremap k nzz " Next find (swap n)
nnoremap l k " Up (swap k)

" Visual mode remappings
vnoremap J E " Jump end WORD (swap E)
vnoremap j e " Jump end word (swap e)
vnoremap N K " Help (swap K)
vnoremap n j " Down (swap j)
vnoremap E J " Join lines (swap J)
vnoremap e l " Right (swap l)
vnoremap K Nzz " Previous find (swap N)
vnoremap k nzz " Next find (swap n)
vnoremap l k " Up (swap k)
